 14class BasicAuth:
 15    """
 16    This class handles basic authentication for OpenAPI services.
 17    It supports multiple users with hashed passwords stored in a local file.
 18    
 19    https://swagger.io/docs/specification/2-0/authentication/basic-authentication/
 20    """
 21    CONFIG_ATTRIBUTE_AUTH = 'cloudmesh.ai.openapi.authentication'
 22    CONFIG_VALUE_AUTH = 'basic'
 23    HALT_FLAG = '#### basic_auth functionality added'
 24    USERS_FILE = Path(path_expand('~/.cloudmesh/.auth_users.json'))
 25
 26    AUTH_FUNC_TEMPLATE = textwrap.dedent("""
 27    from cloudmesh.ai.openapi.authentication.basic import BasicAuth
 28
 29    def basic_auth(username, password, required_scopes=None):
 30        return BasicAuth.basic_auth(username, password)
 31    
 32    {halt_flag}
 33    """)
 34
 35    @classmethod
 36    def basic_auth(cls, username, password, required_scopes=None):
 37        """
 38        basic_auth function to be listed as x-basicInfoFunc in generated openapi yaml
 39        """
 40        try:
 41            if not cls.USERS_FILE.exists():
 42                return None
 43            
 44            with open(cls.USERS_FILE, "r") as f:
 45                users = json.load(f)
 46            
 47            if username in users:
 48                stored_hash = users[username]['hash']
 49                salt = users[username]['salt']
 50                if cls._hash_password(password, salt) == stored_hash:
 51                    return {'sub': username, 'scope': users[username].get('scope', '')}
 52        except Exception as e:
 53            Console.error(f"Auth error: {e}")
 54            
 55        return None
 56
 57    @classmethod
 58    def reset_users(cls):
 59        """
 60        DEPRECATED
 61        """
 62        pass

 64    @classmethod
 65    def add_user(cls, user, password, scope=""):
 66        """
 67        Add a user with a hashed password to the auth file.
 68        """
 69        import os
 70        salt = os.urandom(16).hex()
 71        hashed = cls._hash_password(password, salt)
 72        
 73        users = {}
 74        if cls.USERS_FILE.exists():
 75            with open(cls.USERS_FILE, "r") as f:
 76                try:
 77                    users = json.load(f)
 78                except json.JSONDecodeError:
 79                    pass
 80        
 81        users[user] = {'hash': hashed, 'salt': salt, 'scope': scope}
 82        
 83        cls.USERS_FILE.parent.mkdir(parents=True, exist_ok=True)
 84        with open(cls.USERS_FILE, "w") as f:
 85            json.dump(users, f, indent=4)
 86        
 87        Console.ok(f"User {user} added successfully")
 88
 89    @classmethod
 90    def _hash_password(cls, password, salt):
 91        """
 92        Hash password using SHA256 and a salt.
 93        """
 94        return hmac.new(salt.encode(), password.encode(), hashlib.sha256).hexdigest()
 95
 96    @classmethod
 97    def write_basic_auth(cls, filename, module_name):
 98        """
 99        Writes the basic auth configuration to a new python file and returns the new module name
100        and new filename
101        """
102        basic_auth_enabled = False
103        for line in open(filename):
104            if cls.HALT_FLAG in line:
105                basic_auth_enabled = True
106                break
107
108        if not basic_auth_enabled:
109            filename_auth = filename[:-len('.py')] + '_basic_auth_enabled.py'
110            copyfile(filename, filename_auth)
111            Console.info(f'copied {filename} to {filename_auth}')
112            filename = filename_auth
113            module_name = module_name + '_basic_auth_enabled'
114            with open(filename, 'a') as f:
115                f.write('\n')
116                f.write(cls.AUTH_FUNC_TEMPLATE.format(halt_flag=cls.HALT_FLAG))
117                f.close()
118            Console.info(f'added basic auth functionality to {filename}')
119
120        return module_name, filename
