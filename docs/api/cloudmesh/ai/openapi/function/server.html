133        """

149    def start(self, name: Optional[str] = None, spec: Optional[str] = None, foreground: bool = False) -> Optional[int]:
150        """
151        Start up an OpenApi server
152
153        :param name:  server name
154        :param spec:  openapi spec yaml file name
155        :param foreground:  flag to run server in foreground.  Default is False.
156        :return: started server PID
157        """
158        name = Server.get_name(name, spec)
159        for active_server in Server.ps():
160            if active_server['name'] == name and os.getpid() != active_server['pid']:
161                Console.error(f'Server {name} already running on PID {active_server["pid"]}')
162                return None
163
164        if foreground:
165            self._run_app()
166            return None
167
168        # Run as background process using subprocess
169        # We create a temporary script to run the server, similar to run_os but cleaner
170        pid = self.run_os()
171        
172        # The run_os method already handles registry registration and printing
173        return pid

175    @staticmethod
176    def ps(name: Optional[str] = None) -> List[Dict[str, Any]]:
177        """
178        List all of the actively running servers using PID files.
179
180        :param name:  Optional server name.
181        :return:  list of running servers.
182        """
183        pids = []
184        # Scan for .pid files in the current directory and subdirectories
185        # In a real production environment, this would be a specific /var/run/cloudmesh directory
186        for pid_file in Path(".").rglob("*.pid"):
187            try:
188                with open(pid_file, "r") as f:
189                    pid = int(f.read().strip())
190                
191                # Verify if the process is actually running
192                if Shell.is_pid_running(pid):
193                    server_name = pid_file.stem
194                    # We can't easily get the spec path from just the PID file 
195                    # unless we store it there, so we'll look it up in the registry
196                    registry = Registry()
197                    entry = registry.list(name=server_name)
198                    spec_path = entry[0]['spec'] if entry else "unknown"
199                    
200                    if name is None or name == server_name:
201                        pids.append({"name": server_name, "pid": pid, "spec": spec_path})
202            except (ValueError, IOError, IndexError):
203                continue
204        
205        # Fallback to Shell.ps() if no PID files found or to complement them
206        if not pids:
207            result = Shell.ps()
208            for pinfo in result:
209                if pinfo["cmdline"]:
210                    line = ' '.join(pinfo["cmdline"])
211                    if "openapi server start" in line:
212                        try:
213                            parts = line.split("start")
214                            spec_path = parts[1].split("--")[0].strip()
215                            server_name = os.path.basename(spec_path).replace(".yaml", "")
216                            if name is None or name == server_name:
217                                pids.append({"name": server_name, "pid": pinfo['pid'], "spec": spec_path})
218                        except (IndexError, ValueError):
219                            continue
220        return pids

234    def stop(name: Optional[str] = None) -> None:
235        """
236        Stop a running OpenApi server
237
238        :param name:  server name
239        """
240        if not name:
241            Console.error("Server name is required to stop a server")
242            return
243
244        Console.ok(f"shutting down server {name}")
245        registry = Registry()
246        entries = registry.list(name=name)
247        
248        pid = None
249        if len(entries) > 1:
250            Console.error(f"Aborting, returned more than one entry from the Registry with the name {name}")
251            return
252        elif len(entries) == 1:
253            pid = str(entries[0]['pid'])
254        else:
255            result = Server.ps(name=name)
256            if result:
257                pid = str(result[0]["pid"])
258
259        if pid:
260            try:
261                Console.info(f"Killing process {pid}")
262                Shell.kill_pid(pid)
263                registry.delete(name=name)
264            except Exception as e:
265                Console.error(f"Failed to kill process {pid}: {e}")
266        else:
267            Console.error(f"No Cloudmesh OpenAPI Server found with the name {name}")

271        """
272        Start an openapi server by creating a physical flask script
273
274        :return:  pid for started server
275        """
276        Console.ok("starting server")
277
278        today_dt = date.today().strftime("%m%d%Y")
279        VERBOSE(f"spec path: {self.spec}")
280
281        flask_script = textwrap.dedent(f'''
282                import connexion
283                import uvicorn
284
285                # Create the application instance with FastAPI backend
286                app = connexion.App(__name__, 
287                                    specification_dir='{self.directory}',
288                                    backend='fastapi')
289
290                # Read the yaml file to configure the endpoints
291                app.add_api('{self.spec}')
292
293                # Add health check endpoint
294                @app.app.get("/health")
295                async def health_check():
296                    return {{"status": "healthy", "server": "{self.name}"}}
297
298                if __name__ == '__main__':
299                    uvicorn.run(app, host='{self.host}', 
300                                port={self.port}, 
301                                log_level='info' if {self.debug} else 'warning')
302            ''')
303
304        script_path = Path(self.directory) / f"{self.name}_cmsoaserver.py"
305        VERBOSE(f"server script: {script_path}")
306
307        try:
308            with open(script_path, 'w') as f:
309                f.write(flask_script)
310        except IOError as e:
311            Console.error(f"Unable to write server file: {e}")
312            raise
313
314        pid = 0
315        try:
316            logname = Path(self.directory) / f"{self.name}_server.{today_dt}.log"
317            with open(logname, "w") as log_file:
318                process = subprocess.Popen([sys.executable, str(script_path)],
319                                           stdout=log_file,
320                                           stderr=subprocess.STDOUT,
321                                           shell=False)
322                pid = process.pid
323            
324            # Write PID file for robust process tracking
325            pid_file = self._get_pid_file(self.name)
326            with open(pid_file, "w") as f:
327                f.write(str(pid))
328                
329        except Exception as e:
330            Console.error(f"Unable to start server: {e}")
331            raise
332
333        try:
334            with open(self.spec, "r") as stream:
335                details = yaml.safe_load(stream)
336        except (yaml.YAMLError, IOError) as e:
337            Console.error(f"Yaml file has syntax error or is missing: {e}")
338            raise
339
340        url = details["servers"][0]["url"]
341
342        print(f"\n   Starting: {self.name}")
343        print(f"   PID:       {pid}")
344        print(f"   Spec:      {self.spec}")
345        print(f"   URL:       {url}\n")
346
347        registry = Registry()
348        ai_metadata = self._get_ai_metadata()
349        
350        registry.add_form_file(details,
351                                pid=pid,
352                                spec=self.spec,
353                                directory=self.directory,
354                                port=self.port,
355                                host=self.host,
356                                url=url,
357                                **ai_metadata
358                                )
359
360        return pid

178        """
179        List all of the actively running servers using PID files.
180
181        :param name:  Optional server name.
182        :return:  list of running servers.
183        """
184        pids = []
185        # Scan for .pid files in the current directory and subdirectories
186        # In a real production environment, this would be a specific /var/run/cloudmesh directory
187        for pid_file in Path(".").rglob("*.pid"):
188            try:
189                with open(pid_file, "r") as f:
190                    pid = int(f.read().strip())
191                
192                # Verify if the process is actually running
193                if Shell.is_pid_running(pid):
194                    server_name = pid_file.stem
195                    # We can't easily get the spec path from just the PID file 
196                    # unless we store it there, so we'll look it up in the registry
197                    registry = Registry()
198                    entry = registry.list(name=server_name)
199                    spec_path = entry[0]['spec'] if entry else "unknown"
200                    
201                    if name is None or name == server_name:
202                        pids.append({"name": server_name, "pid": pid, "spec": spec_path})
203            except (ValueError, IOError, IndexError):
204                continue
205        
206        # Fallback to Shell.ps() if no PID files found or to complement them
207        if not pids:
208            result = Shell.ps()
209            for pinfo in result:
210                if pinfo["cmdline"]:
211                    line = ' '.join(pinfo["cmdline"])
212                    if "openapi server start" in line:
213                        try:
214                            parts = line.split("start")
215                            spec_path = parts[1].split("--")[0].strip()
216                            server_name = os.path.basename(spec_path).replace(".yaml", "")
217                            if name is None or name == server_name:
218                                pids.append({"name": server_name, "pid": pinfo['pid'], "spec": spec_path})
219                        except (IndexError, ValueError):
220                            continue
221        return pids
